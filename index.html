const non = document.getElementById("non"); const card = document.querySelector(".card"); const oui = document.querySelector(".primary"); function initPosition(){ const cardRect = card.getBoundingClientRect(); const ouiRect = oui.getBoundingClientRect(); const nonRect = non.getBoundingClientRect(); const padding = 10; // 🔥 position sous OUI (avec marge + sécurité) const x = (cardRect.width / 2) - (nonRect.width / 2); const y = oui.offsetTop + ouiRect.height + 25; non.style.left = x + "px"; non.style.top = y + "px"; } function teleport(){ const cardRect = card.getBoundingClientRect(); const nonRect = non.getBoundingClientRect(); const padding = 10; const maxX = cardRect.width - nonRect.width - padding; const maxY = cardRect.height - nonRect.height - padding; const x = padding + Math.random() * maxX; const y = padding + Math.random() * maxY; non.style.left = x + "px"; non.style.top = y + "px"; } // ⚡ évite le clic direct sur OUI non.addEventListener("pointerdown", (e) => { e.preventDefault(); e.stopPropagation(); teleport(); }); // ⏳ IMPORTANT : attendre que tout soit rendu window.addEventListener("load", initPosition); window.addEventListener("resize", initPosition);
